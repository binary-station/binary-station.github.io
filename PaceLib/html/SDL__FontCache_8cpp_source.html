194static void FC_MapFree(FC_Map *map) {
195 int i;
196 if (map == nullptr)
197 return;
198
199 // Go through each bucket
200 for (i = 0; i < map->num_buckets; ++i) {
201 // Delete the nodes in order
202 FC_MapNode *node = map->buckets[i];
203 while (node != nullptr) {
204 FC_MapNode *last = node;
205 node = node->next;
206 free(last);
207 }
208 }
209
210 free(map->buckets);
211 free(map);
212}

215static FC_GlyphData *FC_MapInsert(FC_Map *map, Uint32 codepoint,
216 FC_GlyphData glyph) {
217 Uint32 index;
218 FC_MapNode *node;
219 if (map == nullptr)
220 return nullptr;
221
222 // Get index for bucket
223 index = codepoint % map->num_buckets;
224
225 // If this bucket is empty, create a node and return its value
226 if (map->buckets[index] == nullptr) {
227 node = map->buckets[index] = (FC_MapNode *)malloc(sizeof(FC_MapNode));
228 node->key = codepoint;
229 node->value = glyph;
230 node->next = nullptr;
231 return &node->value;
232 }
233
234 for (node = map->buckets[index]; node != nullptr; node = node->next) {
235 // Find empty node and add a new one on.
236 if (node->next == nullptr) {
237 node->next = (FC_MapNode *)malloc(sizeof(FC_MapNode));
238 node = node->next;
239
240 node->key = codepoint;
241 node->value = glyph;
242 node->next = nullptr;
243 return &node->value;
244 }
245 }
246
247 return nullptr;

389int U8_strinsert(char *string, int position, const char *source,
390 int max_bytes) {
391 int pos_u8char;
392 int len;
393 int add_len;
394 int ulen;
395 const char *string_start = string;
396
397 if (string == nullptr || source == nullptr)
398 return 0;
399
400 len = strlen(string);
401 add_len = strlen(source);
402 ulen = U8_strlen(string);
403
404 if (position == -1)
405 position = ulen;
406
407 if (position < 0 || position > ulen || len + add_len + 1 > max_bytes)
408 return 0;
409
410 // Move string pointer to the proper position
411 pos_u8char = 0;
412 while (*string != '\0' && pos_u8char < position) {
413 string = (char *)U8_next(string);
414 ++pos_u8char;
415 }
416
417 // Move the rest of the string out of the way
418 memmove(string + add_len, string, len - (string - string_start) + 1);
419
420 // Copy in the new characters
421 memcpy(string, source, add_len);
422
423 return 1;
424}

456static inline SDL_Rect SDL_RectIntersect(SDL_Rect A, SDL_Rect B) {
457 SDL_Rect result;
458 float Amin, Amax, Bmin, Bmax;
459
460 // Horizontal intersection
461 Amin = A.x;
462 Amax = Amin + A.w;
463 Bmin = B.x;
464 Bmax = Bmin + B.w;
465 if (Bmin > Amin)
466 Amin = Bmin;
467 result.x = Amin;
468 if (Bmax < Amax)
469 Amax = Bmax;
470 result.w = Amax - Amin > 0 ? Amax - Amin : 0;
471
472 // Vertical intersection
473 Amin = A.y;
474 Amax = Amin + A.h;
475 Bmin = B.y;
476 Bmax = Bmin + B.h;
477 if (Bmin > Amin)
478 Amin = Bmin;
479 result.y = Amin;
480 if (Bmax < Amax)
481 Amax = Bmax;
482 result.h = Amax - Amin > 0 ? Amax - Amin : 0;
483
484 return result;
485}

533void FC_GetUTF8FromCodepoint(char *result, Uint32 codepoint) {
534 char a, b, c, d;
535
536 if (result == nullptr)
537 return;
538
539 a = (codepoint >> 24) & 0xFF;
540 b = (codepoint >> 16) & 0xFF;
541 c = (codepoint >> 8) & 0xFF;
542 d = codepoint & 0xFF;
543
544 if (a == 0) {
545 if (b == 0) {
546 if (c == 0) {
547 result[0] = d;
548 result[1] = '\0';
549 } else {
550 result[0] = c;
551 result[1] = d;
552 result[2] = '\0';
553 }
554 } else {
555 result[0] = b;
556 result[1] = c;
557 result[2] = d;
558 result[3] = '\0';
559 }
560 } else {
561 result[0] = a;
562 result[1] = b;
563 result[2] = c;

569Uint32 FC_GetCodepointFromUTF8(const char **c, Uint8 advance_pointer) {
570 Uint32 result = 0;
571 const char *str;
572 if (c == nullptr || *c == nullptr)
573 return 0;
574
575 str = *c;
576 if ((unsigned char)*str <= 0x7F)
577 result = *str;
578 else if ((unsigned char)*str < 0xE0) {
579 result |= (unsigned char)(*str) << 8;
580 result |= (unsigned char)(*(str + 1));
581 if (advance_pointer)
582 *c += 1;
583 } else if ((unsigned char)*str < 0xF0) {
584 result |= (unsigned char)(*str) << 16;
585 result |= (unsigned char)(*(str + 1)) << 8;
586 result |= (unsigned char)(*(str + 2));
587 if (advance_pointer)
588 *c += 2;
589 } else {
590 result |= (unsigned char)(*str) << 24;
591 result |= (unsigned char)(*(str + 1)) << 16;
592 result |= (unsigned char)(*(str + 2)) << 8;
593 result |= (unsigned char)(*(str + 3));
594 if (advance_pointer)
595 *c += 3;
596 }
597 return result;
598}

680 if (font == nullptr)
681 return 0;
682
683 SDL_Texture *new_level = SDL_CreateTexture(
684 font->renderer, SDL_PIXELFORMAT_RGBA8888, SDL_TEXTUREACCESS_TARGET,
685 font->height * 12, font->height * 12);

688 !FC_SetGlyphCacheLevel(font, font->glyph_cache_count, new_level)) {
689 SDL_Log("Error: SDL_FontCache ran out of packing space and could not add "

698 // any draw functions

721 SDL_SetRenderDrawColor(font->renderer, 0, 0, 0, 0);
722 SDL_RenderClear(font->renderer);
723 SDL_SetRenderDrawColor(font->renderer, r, g, b, a);
724 SDL_SetRenderTarget(font->renderer, prev_target);
725 if (prev_target) {
726 if (prev_clip_enabled)
727 set_clip(font->renderer, &prev_clip);
728 if (prev_logicalw && prev_logicalh)
729 SDL_RenderSetLogicalSize(font->renderer, prev_logicalw, prev_logicalh);
730 else {
731 SDL_RenderSetViewport(font->renderer, &prev_viewport);
732 SDL_RenderSetScale(font->renderer, prev_scalex, prev_scaley);

734 }
735 }

741 SDL_Surface *data_surface) {
742 if (font == nullptr || data_surface == nullptr)
743 return 0;
744
745 SDL_Texture *new_level;

747 new_level = SDL_CreateTextureFromSurface(font->renderer, data_surface);
748 else {

750 SDL_Renderer *renderer = font->renderer;

752 // Set filter mode for new texture
753 char old_filter_mode[16]; // Save it so we can change the hint value in the
754 // meantime

757 old_filter_hint = "nearest";

759
760 if (FC_GetFilterMode(font) == FC_FILTER_LINEAR)
761 SDL_SetHint(SDL_HINT_RENDER_SCALE_QUALITY, "1");
762 else
763 SDL_SetHint(SDL_HINT_RENDER_SCALE_QUALITY, "0");
764
765 new_level = SDL_CreateTexture(renderer, data_surface->format->format,
766 SDL_TEXTUREACCESS_TARGET, data_surface->w,
767 data_surface->h);
768 SDL_SetTextureBlendMode(new_level, SDL_BLENDMODE_BLEND);
769
770 // Reset filter mode for the temp texture
771 SDL_SetHint(SDL_HINT_RENDER_SCALE_QUALITY, "0");
772
773 {
774 Uint8 r, g, b, a;
775 SDL_Texture *temp = SDL_CreateTextureFromSurface(renderer, data_surface);
776 SDL_Texture *prev_target = SDL_GetRenderTarget(renderer);
777 SDL_Rect prev_clip, prev_viewport;
778 int prev_logicalw, prev_logicalh;
779 Uint8 prev_clip_enabled;
780 float prev_scalex, prev_scaley;
781 // only backup if previous target existed (SDL will preserve them for the
782 // default target)
783 if (prev_target) {
784 prev_clip_enabled = has_clip(renderer);
785 if (prev_clip_enabled)
786 prev_clip = get_clip(renderer);
787 SDL_RenderGetViewport(renderer, &prev_viewport);
788 SDL_RenderGetScale(renderer, &prev_scalex, &prev_scaley);
789 SDL_RenderGetLogicalSize(renderer, &prev_logicalw, &prev_logicalh);
790 }
791 SDL_SetTextureBlendMode(temp, SDL_BLENDMODE_NONE);
792 SDL_SetRenderTarget(renderer, new_level);
793
794 SDL_GetRenderDrawColor(renderer, &r, &g, &b, &a);
795 SDL_SetRenderDrawColor(renderer, 0, 0, 0, 0);
796 SDL_RenderClear(renderer);
797 SDL_SetRenderDrawColor(renderer, r, g, b, a);
798
799 SDL_RenderCopy(renderer, temp, nullptr, nullptr);
800 SDL_SetRenderTarget(renderer, prev_target);
801 if (prev_target) {
802 if (prev_clip_enabled)
803 set_clip(renderer, &prev_clip);
804 if (prev_logicalw && prev_logicalh)
805 SDL_RenderSetLogicalSize(renderer, prev_logicalw, prev_logicalh);
806 else {
807 SDL_RenderSetViewport(renderer, &prev_viewport);
808 SDL_RenderSetScale(renderer, prev_scalex, prev_scaley);
809 }
810 }
811
812 SDL_DestroyTexture(temp);
813 }
814
815 // Reset to the old filter value
816 SDL_SetHint(SDL_HINT_RENDER_SCALE_QUALITY, old_filter_mode);
817 }
818
819 if (new_level == nullptr ||
820 !FC_SetGlyphCacheLevel(font, cache_level, new_level)) {
821 SDL_Log("Error: SDL_FontCache ran out of packing space and could not add "
822 "another cache level.\n");
823 SDL_DestroyTexture(new_level);
824
825 return 0;
826 }
827 return 1;
828}

830static FC_GlyphData *FC_PackGlyphData(FC_Font *font, Uint32 codepoint,
831 Uint16 width, Uint16 maxWidth,
832 Uint16 maxHeight) {
833 FC_Map *glyphs = font->glyphs;
834 FC_GlyphData *last_glyph = &font->last_glyph;
835 Uint16 height = font->height + FC_CACHE_PADDING;
836
837 // TAB is special!
838 if (codepoint == '\t') {
839 FC_GlyphData spaceGlyph;
840 FC_GetGlyphData(font, &spaceGlyph, ' ');
841 width = fc_tab_width * spaceGlyph.rect.w;
842 }
843
844 if (last_glyph->rect.x + last_glyph->rect.w + width >=
845 maxWidth - FC_CACHE_PADDING) {
846 if (last_glyph->rect.y + height + height >= maxHeight - FC_CACHE_PADDING) {
847 // Get ready to pack on the next cache level when it is ready
848 last_glyph->cache_level = font->glyph_cache_count;
849 last_glyph->rect.x = FC_CACHE_PADDING;
850 last_glyph->rect.y = FC_CACHE_PADDING;
851 last_glyph->rect.w = 0;
852 return nullptr;
853 } else {
854 // Go to next row
855 last_glyph->rect.x = FC_CACHE_PADDING;
856 last_glyph->rect.y += height;
857 last_glyph->rect.w = 0;
858 }
859 }
860
861 // Move to next space
862 last_glyph->rect.x += last_glyph->rect.w + 1 + FC_CACHE_PADDING;
863 last_glyph->rect.w = width;
864
865 return FC_MapInsert(glyphs, codepoint,
866 FC_MakeGlyphData(last_glyph->cache_level,
867 last_glyph->rect.x, last_glyph->rect.y,
868 last_glyph->rect.w, last_glyph->rect.h));
869}

879Uint8 FC_SetGlyphCacheLevel(FC_Font *font, int cache_level,
880 SDL_Texture *cache_texture) {
881 if (font == nullptr || cache_level < 0)
882 return 0;
883
884 // Must be sequentially added
885 if (cache_level > font->glyph_cache_count + 1)
886 return 0;
887
888 if (cache_level == font->glyph_cache_count) {
889 font->glyph_cache_count++;
890
891 // Grow cache?
892 if (font->glyph_cache_count > font->glyph_cache_size) {
893 // Copy old cache to new one
894 int i;
895 SDL_Texture **new_cache;
896 new_cache = (SDL_Texture **)malloc(font->glyph_cache_count *
897 sizeof(SDL_Texture *));
898 for (i = 0; i < font->glyph_cache_size; ++i)
899 new_cache[i] = font->glyph_cache[i];
900
901 // Save new cache
902 free(font->glyph_cache);
903 font->glyph_cache_size = font->glyph_cache_count;
904 font->glyph_cache = new_cache;
905 }
906 }
907
908 font->glyph_cache[cache_level] = cache_texture;
909 return 1;
910}

925#define FC_LOAD_MAX_SURFACES 10
926
927Uint8 FC_LoadFontFromTTF(FC_Font *font, SDL_Renderer *renderer, TTF_Font *ttf,
928 SDL_Color color) {
929 if (font == nullptr || ttf == nullptr)
930 return 0;
931
932 if (renderer == nullptr)
933 return 0;
934
935 FC_ClearFont(font);
936
937 // Might as well check render target support here
938 SDL_RendererInfo info;
939 SDL_GetRendererInfo(renderer, &info);
940 fc_has_render_target_support = (info.flags & SDL_RENDERER_TARGETTEXTURE);
941
942 font->renderer = renderer;
943
944 font->ttf_source = ttf;
945
946 // font->line_height = TTF_FontLineSkip(ttf);
947 font->height = TTF_FontHeight(ttf);
948 font->ascent = TTF_FontAscent(ttf);
949 font->descent = -TTF_FontDescent(ttf);
950
951 // Some bug for certain fonts can result in an incorrect height.
952 if (font->height < font->ascent - font->descent)
953 font->height = font->ascent - font->descent;
954
955 font->baseline = font->height - font->descent;
956
957 font->default_color = color;
958
959 {
960 SDL_Color white = {255, 255, 255, 255};
961 SDL_Surface *glyph_surf;
962 char buff[5];
963 const char *buff_ptr = buff;
964 const char *source_string;
965 Uint8 packed = 0;
966
967 // Copy glyphs from the surface to the font texture and store the position
968 // data Pack row by row into a square texture Try figuring out dimensions
969 // that make sense for the font size.
970 unsigned int w = font->height * 12;
971 unsigned int h = font->height * 12;
972 SDL_Surface *surfaces[FC_LOAD_MAX_SURFACES];
973 int num_surfaces = 1;
974 surfaces[0] = FC_CreateSurface32(w, h);
975 font->last_glyph.rect.x = FC_CACHE_PADDING;
976 font->last_glyph.rect.y = FC_CACHE_PADDING;
977 font->last_glyph.rect.w = 0;
978 font->last_glyph.rect.h = font->height;
979
980 source_string = font->loading_string;
981 for (; *source_string != '\0'; source_string = U8_next(source_string)) {
982 memset(buff, 0, 5);
983 if (!U8_charcpy(buff, source_string, 5))
984 continue;
985 glyph_surf = TTF_RenderUTF8_Blended(ttf, buff, white);
986 if (glyph_surf == nullptr)
987 continue;
988
989 // Try packing. If it fails, create a new surface for the next cache
990 // level.
991 packed = (FC_PackGlyphData(font, FC_GetCodepointFromUTF8(&buff_ptr, 0),
992 glyph_surf->w, surfaces[num_surfaces - 1]->w,
993 surfaces[num_surfaces - 1]->h) != nullptr);
994 if (!packed) {
995 int i = num_surfaces - 1;
996 if (num_surfaces >= FC_LOAD_MAX_SURFACES) {
997 // Can't do any more!
998 SDL_Log("SDL_FontCache error: Could not create enough cache surfaces "
999 "to fit all of the loading string!\n");
1000 SDL_FreeSurface(glyph_surf);
1001 break;
1002 }
1003
1004 // Upload the current surface to the glyph cache now so we can keep the
1005 // cache level packing cursor up to date as we go.
1006 FC_UploadGlyphCache(font, i, surfaces[i]);
1007 SDL_FreeSurface(surfaces[i]);
1008
1009 SDL_SetTextureBlendMode(font->glyph_cache[i], SDL_BLENDMODE_BLEND);
1010
1011 // Update the glyph cursor to the new cache level. We need to do this
1012 // here because the actual cache lags behind our use of the packing
1013 // above.
1014 font->last_glyph.cache_level = num_surfaces;
1015
1016 surfaces[num_surfaces] = FC_CreateSurface32(w, h);
1017 num_surfaces++;
1018 }
1019
1020 // Try packing for the new surface, then blit onto it.
1021 if (packed ||
1022 FC_PackGlyphData(font, FC_GetCodepointFromUTF8(&buff_ptr, 0),
1023 glyph_surf->w, surfaces[num_surfaces - 1]->w,
1024 surfaces[num_surfaces - 1]->h) != nullptr) {
1025 SDL_SetSurfaceBlendMode(glyph_surf, SDL_BLENDMODE_NONE);
1026 SDL_Rect srcRect = {0, 0, glyph_surf->w, glyph_surf->h};
1027 SDL_Rect destrect = font->last_glyph.rect;
1028 SDL_BlitSurface(glyph_surf, &srcRect, surfaces[num_surfaces - 1],
1029 &destrect);
1030 }
1031
1032 SDL_FreeSurface(glyph_surf);
1033 }
1034
1035 {
1036 int i = num_surfaces - 1;
1037 FC_UploadGlyphCache(font, i, surfaces[i]);
1038 SDL_FreeSurface(surfaces[i]);
1039 SDL_SetTextureBlendMode(font->glyph_cache[i], SDL_BLENDMODE_BLEND);
1040 }
1041 }
1042
1043 return 1;
1044}

1266Uint8 FC_GetGlyphData(FC_Font *font, FC_GlyphData *result, Uint32 codepoint) {
1267 FC_GlyphData *e = FC_MapFind(font->glyphs, codepoint);
1268 if (e == nullptr) {
1269 char buff[5];
1270 int w, h;
1271 SDL_Color white = {255, 255, 255, 255};
1272 SDL_Surface *surf;
1273 SDL_Texture *cache_image;
1274
1275 if (font->ttf_source == nullptr)
1276 return 0;
1277
1278 FC_GetUTF8FromCodepoint(buff, codepoint);
1279
1280 cache_image = FC_GetGlyphCacheLevel(font, font->last_glyph.cache_level);
1281 if (cache_image == nullptr) {
1282 SDL_Log("SDL_FontCache: Failed to load cache image, so cannot add new "
1283 "glyphs!\n");
1284 return 0;
1285 }
1286
1287 SDL_QueryTexture(cache_image, nullptr, nullptr, &w, &h);
1288
1289 surf = TTF_RenderUTF8_Blended(font->ttf_source, buff, white);
1290 if (surf == nullptr) {
1291 return 0;
1292 }
1293
1294 e = FC_PackGlyphData(font, codepoint, surf->w, w, h);
1295 if (e == nullptr) {
1296 // Grow the cache
1297 FC_GrowGlyphCache(font);
1298
1299 // Try packing again
1300 e = FC_PackGlyphData(font, codepoint, surf->w, w, h);
1301 if (e == nullptr) {
1302 SDL_FreeSurface(surf);
1303 return 0;
1304 }
1305 }
1306
1307 // Render onto the cache texture
1308 FC_AddGlyphToCache(font, surf);
1309
1310 SDL_FreeSurface(surf);
1311 }
1312
1313 if (result != nullptr && e != nullptr)
1314 *result = *e;
1315
1316 return 1;
1317}

1325static SDL_Rect FC_RenderLeft(FC_Font *font, SDL_Renderer *dest, int x, int y,
1326 FC_Scale scale, const char *text) {
1327 const char *c = text;
1328 SDL_Rect srcRect;
1329 SDL_Rect dstRect;
1330 SDL_Rect dirtyRect = {x, y, 0, 0};
1331
1332 FC_GlyphData glyph;
1333 Uint32 codepoint;
1334
1335 float destX = x;
1336 float destY = y;
1337 float destH;
1338 float destLineSpacing;
1339 float destLetterSpacing;
1340
1341 if (font == nullptr)
1342 return dirtyRect;
1343
1344 destH = font->height * scale.y;
1345 destLineSpacing = font->lineSpacing * scale.y;
1346 destLetterSpacing = font->letterSpacing * scale.x;
1347
1348 if (c == nullptr || font->glyph_cache_count == 0 || dest == nullptr)
1349 return dirtyRect;
1350
1351 int newlineX = x;
1352
1353 for (; *c != '\0'; c++) {
1354 if (*c == '\n') {
1355 destX = newlineX;
1356 destY += destH + destLineSpacing;
1357 continue;
1358 }
1359
1360 codepoint = FC_GetCodepointFromUTF8(
1361 &c, 1); // Increments 'c' to skip the extra UTF-8 bytes
1362 if (!FC_GetGlyphData(font, &glyph, codepoint)) {
1363 codepoint = ' ';
1364 if (!FC_GetGlyphData(font, &glyph, codepoint))
1365 continue; // Skip bad characters
1366 }
1367
1368 if (codepoint == ' ') {
1369 destX += glyph.rect.w * scale.x + destLetterSpacing;
1370 continue;
1371 }
1372 /*if(destX >= dest->w)
1373 continue;
1374 if(destY >= dest->h)
1375 continue;*/
1376
1377 srcRect = glyph.rect;
1378
1379 dstRect =
1380 fc_render_callback(FC_GetGlyphCacheLevel(font, glyph.cache_level),
1381 &srcRect, dest, destX, destY, scale.x, scale.y);
1382 if (dirtyRect.w == 0 || dirtyRect.h == 0)
1383 dirtyRect = dstRect;
1384 else
1385 dirtyRect = SDL_RectUnion(dirtyRect, dstRect);
1386
1387 destX += glyph.rect.w * scale.x + destLetterSpacing;
1388 }
1389
1390 return dirtyRect;
1391}

1472static FC_StringList *FC_Explode(const char *text, char delimiter) {
1473 FC_StringList *head;
1474 FC_StringList *new_node;
1475 FC_StringList **node;
1476 const char *start;
1477 const char *end;
1478 unsigned int size;
1479 if (text == nullptr)
1480 return nullptr;
1481
1482 head = nullptr;
1483 node = &head;
1484
1485 // Doesn't technically support UTF-8, but it's probably fine, right?
1486 size = 0;
1487 start = end = text;
1488 while (1) {
1489 if (*end == delimiter || *end == '\0') {
1490 *node = (FC_StringList *)malloc(sizeof(FC_StringList));
1491 new_node = *node;
1492
1493 new_node->value = (char *)malloc(size + 1);
1494 memcpy(new_node->value, start, size);
1495 new_node->value[size] = '\0';
1496
1497 new_node->next = nullptr;
1498
1499 if (*end == '\0')
1500 break;
1501
1502 node = &((*node)->next);
1503 start = end + 1;
1504 size = 0;
1505 } else
1506 ++size;
1507
1508 ++end;
1509 }
1510
1511 return head;
1512}

1604static FC_StringList *FC_GetBufferFitToColumn(FC_Font *font, int width,
1605 FC_Scale scale,
1606 Uint8 keep_newlines) {
1607 FC_StringList *result = nullptr;
1608 FC_StringList **current = &result;
1609
1610 FC_StringList *ls, *iter;
1611
1612 ls = (keep_newlines ? FC_ExplodeAndKeep(fc_buffer, '\n')
1613 : FC_Explode(fc_buffer, '\n'));
1614 for (iter = ls; iter != nullptr; iter = iter->next) {
1615 char *line = iter->value;
1616
1617 // If line is too long, then add words one at a time until we go over.
1618 if (width > 0 && FC_GetWidth(font, "%s", line) > width) {
1619 FC_StringList *words, *word_iter, *spaces, *spaces_iter;
1620
1621 words = FC_ExplodeBreakingSpace(line, &spaces);
1622 // Skip the first word for the iterator, so there will always be at least
1623 // one word per line
1624 line = new_concat(words->value, spaces->value);
1625 for (word_iter = words->next, spaces_iter = spaces->next;
1626 word_iter != nullptr && spaces_iter != nullptr;
1627 word_iter = word_iter->next, spaces_iter = spaces_iter->next) {
1628 char *line_plus_word = new_concat(line, word_iter->value);
1629 char *word_plus_space =
1630 new_concat(word_iter->value, spaces_iter->value);
1631 if (FC_GetWidth(font, "%s", line_plus_word) > width) {
1632 current = FC_StringListPushBack(current, line, 0);
1633
1634 line = word_plus_space;
1635 } else {
1636 replace_concat(&line, word_plus_space);
1637 free(word_plus_space);
1638 }
1639 free(line_plus_word);
1640 }
1641 current = FC_StringListPushBack(current, line, 0);
1642 FC_StringListFree(words);
1643 FC_StringListFree(spaces);
1644 } else {
1645 current = FC_StringListPushBack(current, line, 0);
1646 iter->value = nullptr;
1647 }
1648 }
1649 FC_StringListFree(ls);
1650
1651 return result;
1652}
